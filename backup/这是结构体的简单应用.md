//这是结构体的简单应用

//这样写好像是作用全局
#include <stdio.h>
#include <string.h>

struct nigger
{
    int age;
    char gongzhuo[20]; // 修改为字符数组以存储字符串
};

struct nigger LiTianShuo;

int main()
{
    struct nigger LiTianShuo = { 24, "student" }; // 正确初始化字符串

    // 修正printf格式字符串和参数
    printf("%d\n%s\n", LiTianShuo.age, LiTianShuo.gongzhuo);

    return 0;
}
